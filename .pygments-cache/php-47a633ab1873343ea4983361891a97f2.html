/**
     * Decode from bas64 if it actually is base64 encoded in the first place
     *
     * If not return the raw string
     *
     * @param $matches
     * @return string
     */
    protected function _decode($matches)
    {
        $raw=$matches[1];
        $decoded = base64_decode($raw, true);
        $return=$decoded;
        if(false === $decoded){
            $return=$raw;
        }elseif(base64_encode($decoded) != $raw){
            $return=$raw;
        }
        return '><![CDATA[' . $return . ']]><';
    }
