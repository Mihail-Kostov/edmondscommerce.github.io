public function getConfigurableProduct() {
	switch ($this->type_id) {
		case 'configurable':
			return $this;
			break;
		default:
			$configurable_product_model = Mage::getModel('catalog/product_type_configurable');
			$parentId= $configurable_product_model->getParentIdsByChild($this->getId());
			if(isset($parentId[0])) {
				$this->load($parentId[0]);
			}
			break;
	}

	return $this;
}
