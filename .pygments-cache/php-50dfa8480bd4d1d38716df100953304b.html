function ec_exceptions_handler(Exception $e) {
    ob_start();
    h(1, 'Uncaught Exception');
    h(2, $e->getMessage());
    h(3, 'Line ' . $e->getLine() . ' of ' . $e->getFile());
    foreach ($e->getTrace() as $t) {
        var_dump($t);
    }
    dumpObjects();
    $exceptionMessage = ob_get_clean();
    echo $exceptionMessage;
    exit(1);
}
