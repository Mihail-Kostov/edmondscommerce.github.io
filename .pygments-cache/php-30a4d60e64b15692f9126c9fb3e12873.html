public function _construct(){
        parent::_construct();
        $this->addData(array(
            'cache_lifetime' => 999999999,
            'cache_tags' => array(Mage_Catalog_Model_Product::CACHE_TAG, Mage_Catalog_Model_Category::CACHE_TAG),
            'cache_key' => Mage::helper('blockcaching')->ecCacheKey($this)
        ));

    }
